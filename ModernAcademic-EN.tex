An inline example: the signal-to-noise ratio may be written as $\mathrm{SNR}=10\log_{10}(P_{\mathrm{sig}}/P_{\mathrm{noise}})$. For typesetting units use the `siunitx` package: e.g. \SI{3.3}{\kilo\byte} or \SI{2e3}{\metre\per\second}.

Displayed equations and alignment:
\begin{equation}
	E = mc^2 \label{eq:energy}
\end{equation}

Multiple aligned equations and a referenced equation:
\begin{align}
	H(X)                  & = -\sum_{x} p(x) \log p(x), \label{eq:entropy}     \\
	D_{\mathrm{KL}}(P\|Q) & = \sum_x P(x) \log\frac{P(x)}{Q(x)}. \label{eq:kl}
\end{align}

We can refer to Equation~\eqref{eq:entropy} using \verb|\eqref| or \verb|\cref|: see \cref{eq:entropy,eq:kl}.

\begin{theorem}[Sample theorem]
	\label{thm:sample}
	Let $f:\mathbb{R}\to\mathbb{R}$ be continuous and integrable on $[a,b]$. Then
	\[\int_a^b f(x)\,dx = F(b)-F(a)\]
	where $F$ is any antiderivative of $f$.
\end{theorem}

\begin{proof}
	This is the Fundamental Theorem of Calculus; a concise proof appears in many analysis texts \cite{berger1971rate}.
\end{proof}

\section{Algorithms and Code}
\label{sec:algo}

Pseudocode using the `algorithm` + `algpseudocode` packages:
\begin{algorithm}
	\caption{Simple gradient descent}
	\label{alg:gd}
	\begin{algorithmic}[1]
		\Procedure{GradientDescent}{$f, x_0, \eta, T$}
		\State $x \gets x_0$
		\For{$t\gets 1$ to $T$}
		\State $g \gets \nabla f(x)$
		\State $x \gets x - \eta g$
		\EndFor
		\State \textbf{return} $x$
		\EndProcedure
	\end{algorithmic}
\end{algorithm}
